31void CareSchedulerObserver::onLifeCycleChanged(Plant *plantTesting,

33 // std::cout << "[CareScheduler] State changed for " << plant->name( << "\n";
34 if (!plantTesting || !handler_ || !newState)
35 return;
36
37 Commands *command = nullptr;
38 if (dynamic_cast<SeedlingState *>(newState)) {
39 command = new SeedlingStateCmd();
40 } else if (dynamic_cast<GrowingState *>(newState)) {
41 command = new GrowingStateCmd();
42 } else if (dynamic_cast<DormantState *>(newState)) {
43 command = new DormantStateCmd();
44 } else if (dynamic_cast<MatureState *>(newState)) {
45 command = new MatureStateCmd();
46 } else if (dynamic_cast<SellingState *>(newState)) {
47 command = new SellingStateCmd();
48 } else if (dynamic_cast<DeadState *>(newState)) {
49 command = new DeadStateCmd();
50 }
51
52 if (command) {
53 plantTesting->detach(this);
54 handler_->handleRequest(command, plantTesting);
55 plantTesting->attach(this);
56 }
57}
