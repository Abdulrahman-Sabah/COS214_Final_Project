60bool assertState(const std::string& testName, bool condition) {
61 if (condition) {
62 std::cout << ANSI_GREEN << "PASS: " << testName << ANSI_RESET << std::endl;
63 return true;
64 } else {
65 std::cout << ANSI_RED << "FAIL: " << testName << " , Condition failed, it was false\n" << ANSI_RESET;
66 return false;
67 }
68}

71template<typename T>
72bool assertStateEqual(const std::string& testName, const T& expected, const T& actual) {
73 if (expected == actual) {
74 std::cout << ANSI_GREEN << "PASS: " << testName << ANSI_RESET << std::endl;
75 return true;
76 }
77 else {
78 std::cout << ANSI_RED << "FAIL: " << testName << " What is expected: " << expected
79 << ", What we got: " << actual << "]\n" << ANSI_RESET;
80 return false;
81 }
